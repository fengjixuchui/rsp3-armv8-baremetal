112 void SynchronousEL1Handle(uint64_t *savedRegisters)//savedRegisters[31], from X30-X0
113 {
114  kout <<INFO<< "processing SynchronousEL1 \n";
115  auto curEL = RegCurrentEL::read();
116  kout << INFO << "CurrentEL = " << curEL.EL << "\n";
117  uint32_t fromEL = RegSPSR_EL1::read().EL;
118  kout << INFO << "From EL = " << fromEL << "\n";
119 
120  auto far = RegFAR_EL1::read();
121  auto elr = RegELR_EL1::read();
122  auto spsr = RegSPSR_EL1::read();
123  auto esr = RegESR_EL1::read();
124  esr.dump();
125  elr.dump();
126  far.dump();
127  spsr.dump();
128  if(esr.EC == ExceptionClass::UNDEF_INST)
129  {
130  kout << INFO << "Undefined instruction\n";
131  kout << INFO << "not processing it\n";
132  asm_wfe_loop();
133  }
134 
135  else if(esr.EC == ExceptionClass::SVC_AA64)
136  {
137  uint16_t svcNumber = lowerMaskBits(16)&esr.ISS;
138  kout << INFO << "svc targeting number : " << svcNumber<< "\n";
139  if(svcNumber == static_cast<decltype(svcNumber)>(SvcFunc::puts)) // 需要参数，参数保存在哪里？保存在栈上
140  {
141  auto str = reinterpret_cast<const char*>(savedRegisters[0]);
142  auto len = reinterpret_cast<size_t>(savedRegisters[1]);
143  auto printkChars = kout.print(str, len);
144  savedRegisters[0] = printkChars; // savedResult
145  }else if(svcNumber == static_cast<decltype(svcNumber)>(SvcFunc::killProcess)){
146  kout << "killing Process \n";
147  // 收回资源： 占用的内存，占用的pid，打开的文件等， 将其从进程队列中清除
148  auto pid = static_cast<Pid>(savedRegisters[0]);
149  int status = *reinterpret_cast<int*>(savedRegisters+1);
150  (void)status;
151  if(pid == PID_CURRENT)
152  {
153  processManager.killProcess(processManager.currentRunningProcess());
154  processManager.scheduleNextProcess(savedRegisters);
155  }
156  asm_wfe_loop();
157  }else if(svcNumber == static_cast<decltype(svcNumber)>(SvcFunc::fork)){
158  // fork父进程返回子进程pid，子进程返回pid=0, 如果失败，返回PID_INVALID
159  auto cur = processManager.currentRunningProcess();
160  // 更新当前进程的上下文，使其与真实的状态保持对应，这样fork才能正确执行。
161  cur->data<true>().saveContext(savedRegisters);
162  auto forked = processManager.forkProcess(cur);
163  if(forked)
164  {
165  // 将forked进程加入就绪队列, 注意，这里对返回值的设置，由于是当前进程，所以直接设置savedRegisters即可
166  processManager.changeProcessStatus(forked, Process::Status::READY);
167  forked->data<true>().registers()[0] = PID_CURRENT ;
168  savedRegisters[0] = forked->data<true>().pid();
169  }else{
170  savedRegisters[0] = PID_INVALID;
171  }
172  }else if(svcNumber == static_cast<decltype(svcNumber)>(SvcFunc::scheduleNext)){
173  processManager.scheduleNextProcess(savedRegisters);
174  }
175  }else if(esr.EC == ExceptionClass::INSTR_ABORT_LOWER_EL){
176  kout << "Instruction Abort \n";
177  uint64_t ifsc=getBits(esr.ISS,0,5);
178  (void)ifsc;// 7=level3 translation fault
179  // 11=level3 access fault
180  int farNotValid = getBits(esr.ISS,10,10);
181  kout << "farNotValid?" << farNotValid << "\n";
182  if(!farNotValid)
183  {
184  kout << INFO << "FAR = " << RegFAR_EL1::read().faultAddr << "\n";
185  }
186  kout << "not processing it\n";
187  asm_wfe_loop();
188  }else if(esr.EC == ExceptionClass::DATA_ABORT_LOWER_EL){
189  kout << INFO << "Data Abort from lower EL\n";
190  kout << INFO << "not processing it\n";
191  asm_wfe_loop();
192  }else if(esr.EC == ExceptionClass::DATA_ABORT_SAME_EL){
193  kout << INFO << "Data Abort at same EL\n";
194  kout << INFO << "not processing \n";
195  asm_wfe_loop();
196  }else if(esr.EC == ExceptionClass::PC_ALIGNMENT_FAULT){
197  kout << INFO << "PC alignment fault\n";
198  kout << INFO << "not processing it\n";
199  asm_wfe_loop();
200  }else if(esr.EC == ExceptionClass::SP_ALIGNMENT_FAULT){
201  kout << INFO << "SP alignment fault\n";
202  kout << INFO << "not processing it \n";
203  asm_wfe_loop();
204  }else if(esr.EC == ExceptionClass::SERROR_INTERRUPT){
205  kout << INFO << "SError interrupt\n";
206  kout << INFO << "not processing it\n";
207  asm_wfe_loop();
208  }else{
209  kout << INFO << "Currently unhandled exception\n";
210  asm_wfe_loop();
211  }
