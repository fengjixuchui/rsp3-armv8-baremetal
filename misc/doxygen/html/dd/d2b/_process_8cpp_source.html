82 Process::Error Process::setupTables(size_t codeSize,size_t heapSize,size_t spSize)
83 {
84  this->_codeSize = codeSize;
85  this->_codeBase = mman.allocate(codeSize,TABLE_ALIGNMENT);
86  this->_spSize = spSize;
87  this->_spBase = mman.allocate(spSize,TABLE_ALIGNMENT);
88  this->_heapSize = heapSize;
89  this->_heapBase = mman.allocate(heapSize,TABLE_ALIGNMENT);
90 
91  this->_tableL0 = mman.allocateAs<RegDescriptor4KBL0*>(TABLE_SIZE,TABLE_ALIGNMENT);
92  this->_tableL1 = mman.allocateAs<RegDescriptor4KBL1*>(TABLE_SIZE,TABLE_ALIGNMENT);
93  this->_tableL2 = mman.allocateAs<RegDescriptor4KBL2*>(TABLE_SIZE,TABLE_ALIGNMENT);
94  this->_tableL3 = mman.allocateAs<RegDescriptor4KBL3*>(TABLE_SIZE,TABLE_ALIGNMENT);
95 
96  if(_codeBase==nullptr || _heapBase==nullptr|| _spBase==nullptr || _tableL0 == nullptr || _tableL1==nullptr || _tableL2==nullptr || _tableL3==nullptr)
97  {
98  kout << FATAL << "a level of page table,heap or stack can not be allocated\n";
99  return Error::SPACE_ALLCOATE_FAILED;
100  }
101 
102  auto phyaddr = asm_at(reinterpret_cast<uint64_t>(_tableL0));
103  _ttbr0.ASID = _pid;
104  _ttbr0.BADDR = ( (static_cast<uint64_t>(phyaddr.S0.PA51_48) << 48) | (phyaddr.S0.PA47_12 << 12))>>1;
105  _ttbr0.CnP = 0;
106 
107  for(size_t i=0;i!=512;++i)
108  {
109  _tableL0[i].Valid = 0; // invalidate all
110  _tableL1[i].S0.Valid = 0; // invalidate all
111  _tableL2[i].S0.Valid = 0; // invalidate all
112  _tableL3[i].Valid = 0; // invalidate all
113  }
114 
115  _tableL0[0] = {0};
116  _tableL0[0].Valid = 1;
117  _tableL0[0].IsTable = 1;
118  phyaddr = asm_at(reinterpret_cast<uint64_t>(_tableL1));
119  _tableL0[0].NextLevelTableAddr = (static_cast<uint64_t>(phyaddr.S0.PA51_48) << (48-12)) | (phyaddr.S0.PA47_12);
120 
121 
122  _tableL1[0].S1.IsTable = 1;
123  _tableL1[0].S1.Valid = 1;
124  phyaddr = asm_at(reinterpret_cast<uint64_t>(_tableL2));
125  _tableL1[0].S1.NextLevelTableAddr = ( static_cast<uint64_t>(phyaddr.S0.PA51_48) << (48-12)) | (phyaddr.S0.PA47_12);
126 
127 
128  _tableL2[0].S1.IsTable = 1;
129  _tableL2[0].S1.Valid = 1;
130  phyaddr = asm_at(reinterpret_cast<uint64_t>(_tableL3));
131  _tableL2[0].S1.NextLevelTableAddr = (static_cast<uint64_t>(phyaddr.S0.PA51_48) << (48-12)) | (phyaddr.S0.PA47_12);
132 
133  phyaddr = asm_at(reinterpret_cast<uint64_t>(_codeBase));
134  // 映射代码到4-24KB
135  for(size_t i=0;i!=CODE_L3_ENTRY_NUM;++i)
136  {
137  _tableL3[CODE_L3_INDEX + i] = {0};
138  _tableL3[CODE_L3_INDEX + i].RES1_0 = 1;
139  _tableL3[CODE_L3_INDEX + i].AF = 1;
140  _tableL3[CODE_L3_INDEX + i].Valid = 1;
141  _tableL3[CODE_L3_INDEX + i].nG = 1;
142  _tableL3[CODE_L3_INDEX + i].NS = 1;
143  _tableL3[CODE_L3_INDEX + i].Contiguous = 1;
144  _tableL3[CODE_L3_INDEX + i].AP = 0b11;//RO,EL0
145  _tableL3[CODE_L3_INDEX + i].OutputAddr = ( (static_cast<uint64_t>(phyaddr.S0.PA51_48) << (48-12))|phyaddr.S0.PA47_12 )+i;
146  }
147  auto spPhAddr = asm_at(reinterpret_cast<uint64_t>(_spBase));
148  for(size_t i=0;i<Process::STACK_L3_ENTRY_NUM;++i)
149  {
150  _tableL3[Process::STACK_L3_INDEX+i] = {0};
151  _tableL3[Process::STACK_L3_INDEX+i].RES1_0 = 1;
152  _tableL3[Process::STACK_L3_INDEX+i].AF = 1;
153  _tableL3[Process::STACK_L3_INDEX+i].Valid = 1;
154  _tableL3[Process::STACK_L3_INDEX+i].nG = 1;
155  _tableL3[Process::STACK_L3_INDEX+i].NS = 1;
156  _tableL3[Process::STACK_L3_INDEX+i].Contiguous=0;
157  _tableL3[Process::STACK_L3_INDEX+i].AP=0b01;//RW, EL0
158  _tableL3[Process::STACK_L3_INDEX+i].OutputAddr = ((static_cast<uint64_t>(spPhAddr.S0.PA51_48) << (48-12))|(spPhAddr.S0.PA47_12)) + i;
159 
160  }
161  auto headPhyAddr=asm_at(reinterpret_cast<uint64_t>(_heapBase));
162  for(size_t i=0;i!=Process::HEAP_L3_ENTRY_NUM;++i)
163  {
164  _tableL3[Process::HEAP_L3_INDEX + i]={0};
165  _tableL3[Process::HEAP_L3_INDEX+i].RES1_0 = 1;
166  _tableL3[Process::HEAP_L3_INDEX+i].AF = 1;
167  _tableL3[Process::HEAP_L3_INDEX+i].Valid = 1;
168  _tableL3[Process::HEAP_L3_INDEX+i].nG = 1;
169  _tableL3[Process::HEAP_L3_INDEX+i].NS = 1;

173  }

226 {
227  this->_ttbr0.write();
228  this->_ELR.write();
229  this->_SPSR.write();
230  this->_spEL0.write();
231 
232  // 从低地址开始还原
233  __asm__ __volatile__(
234  "cbz %1, 1f \n\t" // if savedSpEL1==nullptr, branch
235  "mov sp, %1 \n\t" // else set sp=savedSpEL1
236  "1: \n\t"
237  "mov x30, %0 \n\t"
238  RESTORE_REGS_ASM_INSTR_X30_BASE

240  ::"r"(_registers),"r"(savedSpEL1)
241  :"sp"
