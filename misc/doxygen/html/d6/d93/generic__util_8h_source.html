133 AS_MACRO uint64_t middleMaskBits(uint64_t lowerBound,uint64_t upperBound)
134 {
135  // clear lowerBound and then clear upperBound
136  return HEX64(ffff,ffff,ffff,ffff) >> lowerBound << (64 - upperBound - 1 + lowerBound) >> (64-upperBound -1);
