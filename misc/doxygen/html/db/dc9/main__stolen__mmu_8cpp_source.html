48 void mmu_init()
49 {
50  unsigned long r, b;
51  unsigned long *paging=reinterpret_cast<unsigned long*>(_end);
52 
53  /* create MMU translation tables at _end */
54 
55  // TTBR0, identity L1
56  paging[0]=reinterpret_cast<unsigned long>(_end+2*PAGESIZE) | // physical address
57  PT_PAGE | // it has the "Present" flag, which must be set, and we have area in it mapped by pages
58  PT_AF | // accessed flag. Without this we're going to have a Data Abort exception
59  PT_USER | // non-privileged
60  PT_ISH | // inner shareable
61  PT_MEM; // normal memory
62 
63  // identity L2, first 2M block
64  paging[2*512]=reinterpret_cast<unsigned long>(_end+3*PAGESIZE) | // physical address
65  PT_PAGE | // we have area in it mapped by pages
66  PT_AF | // accessed flag
67  PT_USER | // non-privileged
68  PT_ISH | // inner shareable
69  PT_MEM; // normal memory
70 
71  // identity L2 2M blocks
72  b=MMIO_BASE>>21;
73  // skip 0th, as we're about to map it by L3
74  for(r=1;r<512;r++)
75  paging[2*512+r]=reinterpret_cast<unsigned long>((r<<21)) | // physical address
76  PT_BLOCK | // map 2M block
77  PT_AF | // accessed flag
78  PT_NX | // no execute
79  PT_USER | // non-privileged
80  (r>=b? PT_OSH|PT_DEV : PT_ISH|PT_MEM); // different attributes for device memory
81 
82  // identity L3
83  for(r=0;r<512;r++)
84  paging[3*512+r]=reinterpret_cast<unsigned long>(r*PAGESIZE) | // physical address
85  PT_PAGE | // map 4k
86  PT_AF | // accessed flag
87  PT_USER | // non-privileged
88  PT_ISH | // inner shareable
89  ((r<0x80||r>reinterpret_cast<unsigned long>(_data)/PAGESIZE)? PT_RW|PT_NX : PT_RO); // different for code and data
90 
91  // TTBR1, kernel L1
92  paging[512+511]=reinterpret_cast<unsigned long>(_end+4*PAGESIZE) | // physical address
93  PT_PAGE | // we have area in it mapped by pages
94  PT_AF | // accessed flag
95  PT_KERNEL | // privileged
96  PT_ISH | // inner shareable
97  PT_MEM; // normal memory
98 
99  // kernel L2
100  paging[4*512+511]=reinterpret_cast<unsigned long>(_end+5*PAGESIZE) | // physical address
101  PT_PAGE | // we have area in it mapped by pages
102  PT_AF | // accessed flag
103  PT_KERNEL | // privileged
104  PT_ISH | // inner shareable
105  PT_MEM; // normal memory
106 
107  // kernel L3
108  paging[5*512]=static_cast<unsigned long>(MMIO_BASE+0x00201000) | // physical address
109  PT_PAGE | // map 4k
110  PT_AF | // accessed flag
111  PT_NX | // no execute
112  PT_KERNEL | // privileged
113  PT_OSH | // outter shareable
114  PT_DEV; // device memory
115 
116  /* okay, now we have to set system registers to enable MMU */
117 
118  // check for 4k granule and at least 36 bits physical address bus */
119  asm volatile ("mrs %0, id_aa64mmfr0_el1" : "=r" (r));
120  b=r&0xF;
121  if((r&(0xF<<28))/*4k*/ || b<1/*36 bits*/) {
122  kout << FATAL << "ERROR: 4k granule or 36 bit address space not supported\n";
123  return;
124  }
125 
126  // first, set Memory Attributes array, indexed by PT_MEM, PT_DEV, PT_NC in our example
127  r= (0xFF << 0) | // AttrIdx=0: normal, IWBWA, OWBWA, NTR
128  (0x04 << 8) | // AttrIdx=1: device, nGnRE (must be OSH too)
129  (0x44 <<16); // AttrIdx=2: non cacheable
130  asm volatile ("msr mair_el1, %0" : : "r" (r));
131 
132  // next, specify mapping characteristics in translate control register
133  r= (0b00LL << 37) | // TBI=0, no tagging
134  (b << 32) | // IPS=autodetected
135  (0b10LL << 30) | // TG1=4k
136  (0b11LL << 28) | // SH1=3 inner
137  (0b01LL << 26) | // ORGN1=1 write back
138  (0b01LL << 24) | // IRGN1=1 write back
139  (0b0LL << 23) | // EPD1 enable higher half
140  (25LL << 16) | // T1SZ=25, 3 levels (512G)
141  (0b00LL << 14) | // TG0=4k
142  (0b11LL << 12) | // SH0=3 inner
143  (0b01LL << 10) | // ORGN0=1 write back
144  (0b01LL << 8) | // IRGN0=1 write back
145  (0b0LL << 7) | // EPD0 enable lower half
146  (25LL << 0); // T0SZ=25, 3 levels (512G)
147  asm volatile ("msr tcr_el1, %0; isb" : : "r" (r));
148 
149  // tell the MMU where our translation tables are. TTBR_ENABLE bit not documented, but required
150  // lower half, user space
151  asm volatile ("msr ttbr0_el1, %0" : : "r" (reinterpret_cast<unsigned long>(_end) + TTBR_ENABLE));
152  // upper half, kernel space
153  asm volatile ("msr ttbr1_el1, %0" : : "r" (reinterpret_cast<unsigned long>(_end) + TTBR_ENABLE + PAGESIZE));
154 
155  // finally, toggle some bits in system control register to enable page translation
156  asm volatile ("dsb ish; isb; mrs %0, sctlr_el1" : "=r" (r));
157  r|=0xC00800; // set mandatory reserved bits
158  r&=~((1<<25) | // clear EE, little endian translation tables
159  (1<<24) | // clear E0E
160  (1<<19) | // clear WXN
161  (1<<12) | // clear I, no instruction cache
162  (1<<4) | // clear SA0
163  (1<<3) | // clear SA
164  (1<<2) | // clear C, no cache at all
165  (1<<1)); // clear A, no aligment check
166  r|= (1<<0); // set M, enable MMU
167  asm volatile ("msr sctlr_el1, %0; isb" : : "r" (r));
168 
169  ASM_NOP();
170  ASM_NOP();
171  ASM_NOP();
172 
173 }
