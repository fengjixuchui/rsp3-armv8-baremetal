169 MemoryChunk* MemoryChunk::moveAhead(size_t moveSize)
170 {
171  if(this->allocated || moveSize >= this->size)
172  return nullptr;
173  if(moveSize==0)
174  return this;
175  MemoryChunk * nextChunk = reinterpret_cast<MemoryChunk*>(reinterpret_cast<char*>(this) + moveSize);
176  //复制所有的域
177  if(moveSize >= sizeof(MemoryChunk))
178  *nextChunk=*this;
179  else{ // 可能产生冲突
180  MemoryChunk temp=*this;
181  *nextChunk = temp;
182  }
183  nextChunk->size -= moveSize;
184 
185 
186  if(moveSize >= sizeof(MemoryChunk)+1) // 能够形成MemoryChunk
187  {
188  // 更新this
189  this->nextBaseFromEnd=0;//就在End的偏移0处
190  this->size = moveSize - sizeof(MemoryChunk);
191  }else{ // 使用第一个字节记录下一个MemoryChunk所在的位置
192  this->nextValidChunkOffset = moveSize;
193  }

228 void MemoryChunk::mergeTrailingsUnallocated()
229 {
230  if(this->isAllocated() || this->nextBaseFromEnd !=0 || this->isEnd())
231  return;
232  size_t collectedSize = 0;
233  MemoryChunk * freeChunk = reinterpret_cast<MemoryChunk*>(reinterpret_cast<char*>(this)+sizeof(MemoryChunk) + this->size);
234  MemoryChunk * lastChunk = nullptr;
235  while(true)
236  {
237  size_t addingSize=freeChunk->nextValidChunkOffset;
238  if(addingSize==0) // 是一个有效的Chunk
239  {
240  if(!freeChunk->isEnd() && !freeChunk->isAllocated() && freeChunk->nextBaseFromEnd==0) // 连续且未分配的Chunk,且非最后一项
241  addingSize = freeChunk->size + sizeof(MemoryChunk);
242  else // else 第一个已经分配的Chunk，因此合并至多能够到达此处
243  break;
244  }
245  collectedSize += addingSize;
246  lastChunk = freeChunk;
247  freeChunk = reinterpret_cast<MemoryChunk*>(reinterpret_cast<char*>(freeChunk) + addingSize);
248  }
249  // 更新size和nextBase
250  this->size+=collectedSize;
251  if(lastChunk->nextValidChunkOffset==0)
252  this->nextBaseFromEnd = lastChunk->nextBaseFromEnd;
253 }
