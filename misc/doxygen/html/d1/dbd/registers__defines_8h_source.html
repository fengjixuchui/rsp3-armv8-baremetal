70 #define SETUP_REG_HARD_CODED_READ(name) \
71 Reg##name Reg##name::read() \
72 { \
73  Reg##name res; \
74  __asm__ __volatile__( \
75  ".4byte 0b" __stringify(ENCODE_MRS) __stringify(ENCODE_##name) __stringify(ENCODE_X0) "\n\t" \
76  "mov %0,x0 \n\t" \
77  :"=r"(res)); \
78  return res; \
79 }
