25 VirtualAddress& VirtualAddress::index(uint8_t which,uint32_t val)
26 {
27  auto initLevel = initialLevel();
28  if(which < initLevel ) // 不存在这个级别
29  return *this;
30  uint8_t lowerBound = 12 + (3-which)*9;
31  uint8_t upperBound = lowerBound + 8;//正常情况下9位
32  if(which==initLevel)// 更少的位
33  {
34  // example:_addrLen=40bits, upperBound should be lowerBound.
35  upperBound = lowerBound + (_addrLen-rightShiftBits(which))-1;
36  }
37  setBits(_addr, lowerBound, upperBound,val);
38  return *this;
39 }

107 uint8_t VirtualAddress::initialLevel()const
108 {
109  // < 13位 无效
110  // 13位,21位: 3 1-9 0-8
111  // 22位,30位: 2 10-18 9-17
112  // 31位,39位: 1
113  // 40位,48位: 0
114  // > 48位 无效（最多4级)
115  if(_addrLen < 13)
116  return UINT8_MAX;
117  auto tmp = (_addrLen-13)/9;
118  if(tmp > 3)
119  return UINT8_MAX;
120  return 3-tmp;
121 }
