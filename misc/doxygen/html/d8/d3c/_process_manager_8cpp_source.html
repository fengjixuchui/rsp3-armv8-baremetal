33 void ProcessManager::scheduleNextProcess(uint64_t *savedRegsiers)
34 {
35  auto cur = currentRunningProcess();
36 
37  auto nextReady = _statedProcessList[Process::READY].head();
38 
39  if(!nextReady)
40  {
41  if(cur) // 没有下一个就绪的，且有当前进程，则继续运行
42  {
43  kout << "schedule with current continuing\n";
44  cur->data<true>().restoreContextAndExecute();
45  }else{ // 当前没有正在运行的进程，于是只能运行 idle进程(所谓idle进程并不是一个真正的进程，而是表示系统所处的状态：当前没有正在运行的进程， 只能等待某个进程被唤醒。)
46  kout << "schedule with idle \n";
47 
48  // 进入idle的前提条件： RUNNING为空
49  // 无就绪进程 --> 等待中断发生
50  // 中断发生 --> 系统处于idle状态 --> 直接调度就绪的进程
51  // 若无就绪进程，则继续等待
52  asm_wfe_loop();
53  }
54  }else{
55  kout << "schedule with next ready process\n";
56  if(cur) //需要执行切换动作
57  {
58  cur->data<true>().saveContext(savedRegsiers);
59  changeProcessStatus(cur, Process::READY);
60  }
61  changeProcessStatus(nextReady, Process::RUNNING);
62  nextReady->data<true>().restoreContextAndExecute();
63  }
64 }
