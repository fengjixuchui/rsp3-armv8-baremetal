18 bool SDDriver::init()
19 {
20  _transferMode = bitOnes<0,1,4>();
21  _capabilities=reg64<R::reg_capabilities>();
22  _baseClockFreq = getBits(_capabilities, 8, 15)*1000000u;
23  _cardInserted = bitsAnySet<16>(reg32<R::reg_presentstate>());
24  if(!_cardInserted)
25  {
26  return false;
27  }
28  _hostControlVersion = getBits(reg16<R::reg_hostcontrollerver>(), 0, 7);
29  if(_hostControlVersion == HC_VER_3)
30  {
31  reg8<R::reg_powercontrol>()=0x10u;
32  }else{
33  reg8<R::reg_powercontrol>()=0x0u;
34  }
35  const size_t delayTimes=1000*1000;
36  delayCPU(delayTimes);
37 
38 
39  // software reset for all
40  reg8<R::reg_softwarereset>()=0x1u;
41  while(bitsAnySet<0>(reg8<R::reg_softwarereset>()))
42  ;
43 
44  // 选择电压以及启用总线电源
45  if(_hostControlVersion != HC_VER_3)
46  {
47  // D03-D01:0b111, 3v3
48  // D00: 1=Power On
49  reg8<R::reg_powercontrol>()= (0xEu | 0x1u);
50  }else{
51  kout <<FATAL << "designed for version 2 \n";
52  return false;
53  }
54  //设置时钟400 KHz
55  if(!clockFreq(400000u))
56  {
57  return false;
58  }
59 
60  // 选择DMA的模式，设置LED，Speed
61  reg8<R::reg_hostcontrol1>()= 0x00000018u;
62 
63  // 允许所有的中断状态
64  reg16<R::reg_normalintrstsena>()=0x0000FFFFu & (~0x00000100u);
65  reg16<R::reg_errorintrstsena>() = 0x0000F3FFu;
66 
67  // 禁用所有中断信号
68  reg16<R::reg_normalintrsigena>() = 0x0;
69  reg16<R::reg_errorintrsigena>() = 0x0;
70 
71  // 设置block的大小
72  _blockSize = 512u;
73  reg16<R::reg_blocksize>() = _blockSize;
74 
75  // 初始化slave，使其从idle状态进入standby状态
76  if(_hostControlVersion!=HC_VER_3)
77  {
78  // 在发送第一条命令之前，等待足够长的时间
79  delayCPU(delayTimes);
80 
81  // 发送 CMD0，使其从任意状态进入IDLE状态
82  if(!transferCommand(CMD0, 0, 0))return false;
83  // 发送CMD8,确认能够提供的电压
84  // 0x1AA -- 2v7-3v6, AA is check pattern
85  // 如果response=0x1AA,则SD卡版本是v2,否则是v1
86  if(!transferCommand(CMD8, 0x1AAu, 0))return false;
87  auto resp=reg32<R::reg_response0>();
88  _cardVersion = (resp==0x1AA)?(SD_VER_2):(SD_VER_1);
89 
90  resp=0u;
91  // 卡持续保持忙状态，持续发送ACMD41
92  // while(card_power_up_is_busy) ,bit[31]=0==busy,1==complete
93  while(bitsNonSet<31>(resp))
94  {
95  if(!transferCommand(CMD55, 0u, 0u))return false;
96  if(!transferCommand(ACMD41, 0x40000000u | 0x00300000u | (0x1FFu<<15u), 0))return false;
97  resp = reg32<R::reg_response0>();
98  }
99 
100  // CMD2,读取card id
101  if(!transferCommand(CMD2, 0, 0))return false;
102  _cardID[0] = reg64<R::reg_response0>();
103  _cardID[1] = reg64<R::reg_response0 + 4>();
104  _cardID[2] = reg64<R::reg_response0 + 4*2>();
105  _cardID[3] = reg64<R::reg_response0 + 4*3>();
106 
107  // 要求slave响应一个relativeCardAddr
108  do{
109  // CMD3 用于获取relativeCardAddr
110  if(!transferCommand(CMD3, 0, 0))return false;
111  _relativeCardAddr=reg32<R::reg_response0>() & 0xFFFF0000u;
112  }while(_relativeCardAddr==0);
113 
114 
115  // CMD9 获取 CardSpecData
116  if(!transferCommand(SDDriver::CMD9, _relativeCardAddr, 0))return false;
117  _cardSpecData.S0._data[0]=reg32<R::reg_response0>();
118  _cardSpecData.S0._data[1]=reg32<R::reg_response0 + 4>();
119  _cardSpecData.S0._data[2]=reg32<R::reg_response0 + 4*2>();
120  _cardSpecData.S0._data[3]=reg32<R::reg_response0 + 4*3>();
121 
122  if(_cardSpecData.V1.csdVersion==0)
123  {
124  size_t mult= 1<<(_cardSpecData.V1.cSizeMult + 2);
125  size_t blocknr=(_cardSpecData.V1.deviceSize+1) * mult;
126  size_t blocklen=1<<(_cardSpecData.V1.readBlLen);
127  _secCount = blocklen * blocknr / 512;
128  }else{
129  kout << FATAL << "not handle csdVersion!=0\n";
130  return false;
131  }
132 
133  // 发送CMD7选择卡，然后进入transfer_state状态
134  if(!transferCommand(CMD7, _relativeCardAddr, 0))return false;
135 
136  }else{
137  kout << FATAL << "init for HC_VER_3 currently not supported\n";
138  return false;
139  }
140 
141  return true;
142 }

151 SDDriver::CommandStatus SDDriver::clockFreq(uint32_t newFreq)
152 {
153  // 禁用clock
154  auto clk=reg16<R::reg_clockcontrol>();
155  clk &= ~(0x4u | 0x1u);
156  reg16<R::reg_clockcontrol>()=clk;
157 
158  uint16_t divCnt;
159  uint16_t divisor=0u;
160  if(_hostControlVersion!=HC_VER_3)
161  {
162  // 计算除数
163  divCnt=0x1u;
164  while(divCnt <= 256u)
165  {
166  if(_baseClockFreq/divCnt <= newFreq)
167  {
168  divisor = divCnt/2u;
169  break;
170  }
171  divCnt = divCnt << 1u;
172  }
173  if(divCnt > 256u)
174  return false;//failed
175  }
176 
177  // 设置时钟divisor
178  if(_hostControlVersion!=HC_VER_3)
179  {
180  clk = reg16<R::reg_clockcontrol>();
181  clk &= (~0xff00u);
182  divisor = (divisor << 8u);
183  divisor &= 0xff00u;
184 
185  clk |= (divisor | 0x1u);
186  reg16<R::reg_clockcontrol>()=clk;
187  }
188 
189  // 等待直到内部时钟稳定
190  while( (reg16<R::reg_clockcontrol>() & 0x2u) == 0)
191  ;
192 
193  // 启用时钟
194  clk = reg16<R::reg_clockcontrol>();
195  reg16<R::reg_clockcontrol>() = clk | 0x4u;
196 
197  return true;
198 }

287 SDDriver::CommandStatus SDDriver::transferCommand(uint32_t cmd,uint32_t arg,uint32_t blckCnt)
288 {
289  auto & presentReg = reg<uint32_t,reg_presentstate>();
290  CommandStatus status;
291  if( (presentReg & 0x1)!=0U) // !=0
292  return status=false;
293 
294  reg<uint32_t,reg_blockcount>()=static_cast<uint16_t>(blckCnt);
295  reg<uint8_t,reg_timeoutcontrol>()=0xEU;
296  reg<uint32_t,reg_argument1lo>()=arg;
297  reg<uint16_t,reg_normalintrsts>()=0x0000FFFFu;
298  reg<uint16_t,reg_errorintrsts>()=0x0000F3FFu;
299 
300  auto framedCmd = frameCommand(cmd) & 0x3FFFu;
301 
302  if( (cmd != CMD21) && (cmd!=CMD19)){
303  auto tempPstate=presentReg;
304  if(((tempPstate & 0x2u)!= 0U) &&
305  ((framedCmd & 0x20u)!=0U))
306  return false;
307  }
308 
309  reg<uint32_t,reg_transfermode>() = (framedCmd << 16)|_transferMode;
310 
311 
312 
313  /* Polling for response for now */
314  auto& statusReg = reg<uint16_t,reg_normalintrsts>();
315  uint16_t tempStatus=0;
316  do{
317  tempStatus=statusReg;
318  if((cmd==CMD21) || (cmd==CMD19))
319  {
320  if((statusReg & 0x20) != 0U)
321  {
322  statusReg = 0x20;
323  }
324  break;
325  }
326  if((tempStatus & 0x20u)!=0U) // has data ready
327  {
328  status = statusReg;
329  if((static_cast<int32_t>(status) & ~0x1u)==0) // has other status
330  {
331  status = 0x2u; // temp status2,error
332  }
333  /* Write to clear error bits */
334  statusReg = 0x0000F3FFu;
335  return status=true; // TODO: we now thing this is right
336  }
337 
338  }while( (tempStatus & 0x1u) ==0U);
339 
340  reg<uint16_t,reg_normalintrsts>()=0x1u;
341 
342  return status=true;
343 }
